'===========================================================================
' FUNCTION: READ LOGIN DATA FROM EXCEL (LoginData sheet)
' Reads: laneAddress, userId, userPassword, domainName
'===========================================================================
Function ReadLoginData(excelPath, ByRef laneAddress, ByRef userId, ByRef userPassword, ByRef domainName)

    Dim objExcel
    Dim objWorkbook
    Dim objSheet
    Dim lastRow
    Dim i
    Dim paramName
    Dim paramValue

    Set objExcel    = CreateObject("Excel.Application")
    objExcel.Visible = False

    Set objWorkbook = objExcel.Workbooks.Open(excelPath)
    Set objSheet    = objWorkbook.Sheets("LoginData")

    lastRow = objSheet.UsedRange.Rows.Count

    ' LoginData sheet layout:
    ' Column A = ParameterName | Column B = Value
    ' Row 1 = Header, data starts Row 2
    For i = 2 To lastRow

        paramName  = Trim(objSheet.Cells(i, 1).Value)
        paramValue = Trim(objSheet.Cells(i, 2).Value)

        Select Case LCase(paramName)
            Case "laneaddress"  : laneAddress  = paramValue
            Case "userid"       : userId        = paramValue
            Case "userpassword" : userPassword  = paramValue
            Case "domainname"   : domainName    = paramValue
        End Select

    Next

    objWorkbook.Close False   ' Close without saving
    objExcel.Quit

    Reporter.ReportEvent micDone, "LoginData", _
        "Lane: " & laneAddress & " | User: " & userId & " | Domain: " & domainName

End Function

'===========================================================================
Function SaveClipboardToFile()

    Dim html
    Dim clipboardText
    Dim fso
    Dim txtFile

    Set html = CreateObject("htmlfile")
    clipboardText = html.ParentWindow.ClipboardData.GetData("Text")

    If Trim(clipboardText) = "" Then
        Reporter.ReportEvent micFail, "Clipboard Empty", _
            "No text received from remote clipboard."
        Exit Function
    End If

    Set fso = CreateObject("Scripting.FileSystemObject")

    If Not fso.FolderExists("C:\Temp") Then
        fso.CreateFolder "C:\Temp"
    End If

    Set txtFile = fso.CreateTextFile("C:\Temp\RemoteApps.txt", True)

    txtFile.Write clipboardText
    txtFile.Close

    Reporter.ReportEvent micDone, "Clipboard Saved", _
        "Saved to C:\Temp\RemoteApps.txt"

End Function

Function ValidateVersions(excelPath)

    Dim fso
    Dim txtFile
    Dim fileContent

    Set fso = CreateObject("Scripting.FileSystemObject")

    If Not fso.FileExists("C:\Temp\RemoteApps.txt") Then
        Reporter.ReportEvent micFail, "Version File", _
            "RemoteApps.txt not found."
        Exit Function
    End If

    Set txtFile = fso.OpenTextFile("C:\Temp\RemoteApps.txt", 1)
    fileContent = txtFile.ReadAll
    txtFile.Close

    Dim objExcel
    Dim objWorkbook
    Dim objSheet

    Set objExcel = CreateObject("Excel.Application")
    objExcel.Visible = False

    Set objWorkbook = objExcel.Workbooks.Open(excelPath)
    Set objSheet = objWorkbook.Sheets("Versions")

    Dim lastRow
    Dim i

    lastRow = objSheet.Cells(objSheet.Rows.Count, 2).End(-4162).Row

    For i = 2 To lastRow

        Dim appName
        Dim expectedVersion
        Dim actualVersion
        Dim statusValue
        Dim matchedLine
        Dim cmp

        appName = Trim(objSheet.Cells(i, 2).Value)
        expectedVersion = Trim(objSheet.Cells(i, 3).Value)
Reporter.ReportEvent micDone, "Searching", appName
        matchedLine = FindApplication(fileContent, appName)
Reporter.ReportEvent micDone, "Matched Line", matchedLine
        If matchedLine <> "" Then

            actualVersion = GetVersionFromLine(matchedLine)

            If actualVersion <> "NOT FOUND" Then

                cmp = CompareSemanticVersions(actualVersion, expectedVersion)

                If cmp = "EQUAL" Or cmp = "HIGHER" Then
                    statusValue = "PASS"
                Else
                    statusValue = "FAIL"
                End If

            Else

                actualVersion = "NOT FOUND"
                statusValue = "FAIL"

            End If

        Else

            actualVersion = "NOT FOUND"
            statusValue = "FAIL"

        End If

        objSheet.Cells(i, 4).Value = actualVersion
        objSheet.Cells(i, 5).Value = statusValue

        Reporter.ReportEvent micDone, appName, _
            "Expected=" & expectedVersion & _
            " | Actual=" & actualVersion & _
            " | Result=" & statusValue

    Next

    objWorkbook.Save
    objWorkbook.Close
    objExcel.Quit

    Reporter.ReportEvent micDone, _
        "Version Validation", _
        "Completed Successfully"

End Function

Function FindApplication(fileContent, appName)

    Dim arrLines
    Dim lineText
    Dim searchText

    arrLines = Split(fileContent, vbCrLf)

    searchText = LCase(appName)

    If InStr(searchText, "|") > 0 Then
        searchText = Trim(Split(searchText, "|")(UBound(Split(searchText, "|"))))
    End If

    For Each lineText In arrLines

        If InStr(1, LCase(lineText), searchText, vbTextCompare) > 0 Then

            FindApplication = lineText
            Exit Function

        End If

    Next

    FindApplication = ""

End Function
Function GetVersionFromLine(lineText)

    Dim regEx
    Dim matches
Set regEx = CreateObject("VBScript.RegExp")
    
    regEx.Pattern = "([0-9]+(\.[0-9A-Za-z\-]+)+)\s*$"
    regEx.Global = False
    regEx.IgnoreCase = True

    If regEx.Test(lineText) Then

        Set matches = regEx.Execute(lineText)
        GetVersionFromLine = matches(0).Value

    Else

        GetVersionFromLine = "NOT FOUND"

    End If

End Function
'===========================================================================
' FUNCTION: COMPARE SEMANTIC VERSIONS
' Returns: "EQUAL", "HIGHER", or "LOWER"
'===========================================================================
Function CompareSemanticVersions(actualVersion, expectedVersion)

    Dim actualParts
    Dim expectedParts
    Dim maxLen
    Dim i
    Dim actualNum
    Dim expectedNum

    actualParts   = Split(actualVersion,   ".")
    expectedParts = Split(expectedVersion, ".")

    If UBound(actualParts) > UBound(expectedParts) Then
        maxLen = UBound(actualParts)
    Else
        maxLen = UBound(expectedParts)
    End If

    For i = 0 To maxLen

        If i > UBound(actualParts) Then
            actualNum = 0
        Else
            actualNum = CInt(actualParts(i))
        End If

        If i > UBound(expectedParts) Then
            expectedNum = 0
        Else
            expectedNum = CInt(expectedParts(i))
        End If

        If actualNum > expectedNum Then
            CompareSemanticVersions = "HIGHER"
            Exit Function
        ElseIf actualNum < expectedNum Then
            CompareSemanticVersions = "LOWER"
            Exit Function
        End If

    Next

    CompareSemanticVersions = "EQUAL"

End Function
